-- Password Reset Tokens Table
-- Run this SQL script in your database to enable forgot password functionality
-- NOTE: Updated for el1 prefix (El Colt live database)

CREATE TABLE IF NOT EXISTS el1password_reset_tokens (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT UNSIGNED NOT NULL,
    token VARCHAR(255) NOT NULL,
    created_at DATETIME NOT NULL,
    expires_at DATETIME NOT NULL,
    used TINYINT(1) DEFAULT 0,
    INDEX idx_token (token),
    INDEX idx_user_id (user_id),
    INDEX idx_expires_at (expires_at),
    FOREIGN KEY (user_id) REFERENCES el1users(ID) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Verify the table was created
SHOW TABLES LIKE 'el1password_reset_tokens';

-- Check the table structure
DESCRIBE el1password_reset_tokens;
